#Ejemplo de función para calcular el coeficiente de variación
coefficient_of_variation <- function(x) {
  if (length(x) <= 1) {
    stop("Input vector must have more than one element.")
  }
  sd_value <- sd(x)
  mean_value <- mean(x)
  cv <- (sd_value / mean_value) * 100
  return(cv)
}
